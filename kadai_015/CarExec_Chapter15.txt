package kadai_015;

public class CarExec_Chapter15 {
	 // メインメソッドで動作確認
    public static void main(String[] args) {
        Car_Chapter15 car = new Car_Chapter15(); // 初期値：ギア3、速度30km/h
        car.run();  // 現在のギア: 3速、速度: 30km/h

        car.gearChange(5); // 5速に変更
        car.run();  // 現在のギア: 5速、速度: 50km/h

        car.gearChange(3); // 2速に変更
        car.run();  // 現在のギア: 2速、速度: 20km/h

        car.gearChange(7); // 不正なギア
        car.run();  // 現在のギア: 7速、速度: 10km/h
    }
}
